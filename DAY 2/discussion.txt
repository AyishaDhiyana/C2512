#include <iostream>
#include <cmath>

using namespace std;

bool isPrime(int number)
{
    if(number<=0)
    {
        return false; 
    }
    if(number<=3)
    {
        return true;
    }
    if(number % 2 == 0 || number % 3 == 0)
    {
       return false; 
    }
    for(int i=5;i<=sqrt(number);i+6)
    {
        if(number % i ==0)
        {
            return false;
        }
    }
    return true;
}

void isPrimeTest()
{
    int number;
    cout << "Enter a number :";
    cin >> number;
    bool result = isPrime(number);
    
    if(result)
    {
        cout<<"Prime";
    }
    else
    {
        cout<<"Not Prime";
    }
    
}

int main()
{
   isPrimeTest();
   return 0;
}
